#version 330 core

layout (location = 0) in vec3 aPos;
layout (location = 1) in vec2 UV;

out vec2 texCoords;

uniform mat4 translation;
uniform mat4 scale;
uniform mat4 projection;

void main() 
{
	mat4 model = translation * scale;
	gl_Position = projection * model * vec4 (aPos, 1.0);
	texCoords = UV;
}
